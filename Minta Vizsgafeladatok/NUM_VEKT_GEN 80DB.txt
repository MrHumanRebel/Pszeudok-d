/*intervallumban 2-es Seed értékkel, majd határozza meg a generált adatokra az 1000-nél nagyobb */
/*értéku adatok átlagát!*/

OSSZ<-0
DB<-0

NUM_VEKT_GEN(A,80,0,5000,2)

for I<-1, 80
  if A[I] > 1000
    OSSZ<-OSSZ+I
    DB<-DB+1
ATL<-OSSZ/DB
