/*A STR_MTX_GEN szubrutin segítségével generáljon egy 10 soros és 5 oszlopos, 25 karakter hosszú*/
/*sztringeket tartalmazó mátrixt 1-es Seed értékkel, majd rendezze a mátrix sorait a 3. oszlop szerint */
/*növekvo sorrendbe! Adja meg az elso 5 db karakterét annak a sztringnek, amelyik az így eloállt mátrix */
/*1. sorának 5. oszlopába került!*/
/*Mr_HumanRebel*/

STR_MTX_GEN(A,10,5,25,1)

/* A két ciklus a buborékrendezés implementálásához szükséges */
for I <- 1, 9
  for J <- 1, 9 - I
    /* A cserét akkor hajtjuk végre, ha a következo sztring hátrább van, mint kéne */
    if A[J, 3] > A[J + 1, 3]
      /* A K ciklusváltozó segítségével végigmegyünk */
      /* a sorok oszlopain, hogy ki tudjuk cserélni */
      for K <- 1, 5
        CS <- A[J, K]
        A[J, K] <- A[J + 1, K]
        A[J + 1, K] <- CS


ER <- COPY(A[1,5],1,5)
Ki: ER
