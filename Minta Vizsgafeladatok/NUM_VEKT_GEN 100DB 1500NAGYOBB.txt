/*A NUM_VEKT_GEN szubrutin segítségével generáljon 100 darab egész számot*/
/*az [1000,2000] intervallumon 2-es seed értékkel*/
/*majd határozza meg a generált adatokra az 1500-nál nagyobb érétéku adatok összegét*/
/*Mr_HumanRebel*/

SEED<-2
TOL<-1000
IG<-2000
N<-100
OSSZ<-DB<-0

NUM_VEKT_GEN(A,N,TOL,IG,SEED)

for I<-1,N
  if A[I]>1500
    DB<-DB+1
    OSSZ<-OSSZ+A[I]
