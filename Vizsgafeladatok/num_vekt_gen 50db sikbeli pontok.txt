/*NUM_VEKT_GEN-el 50db síkbeli pontot úgy, hogy a koordináták [-1000,1000] intervallumba essenek*/
/*X-eket 1-es seed-el Y-okat 2-el*/
/*Határozza meg az origótól legtávolabb lévo pont távolságát az origótól*/
/*Eredményt egy tizedesjegyre*/
/*Mr_HumanRebel*/

N<-50
TOL<--1000
IG<-1000
XSEED<-1
YSEED<-2

NUM_VEKT_GEN(X,N,TOL,IG,XSEED)
NUM_VEKT_GEN(Y,N,TOL,IG,YSEED)

for I<-1, N
  MTX[I,1]<-X[I]
  MTX[I,2]<-Y[I]

for I<-1, N
  Ki:"X Koordináta: ",MTX[I,1]
  Ki:"Y Koordináta: ",MTX[I,2]
  Ki:" " 

for I<-1,N
  TAV[I]<-SQRT(SQR((MTX[I,1]-0))+SQR((MTX[I,2]-0)))

for I<-1, N
  Ki:"Adott pont távolsága origótól: ",TAV[I]
  Ki:" " 

for I<-1,N-1
  for J<-N,I+1,-1
    if TAV[J]>TAV[J-1]
      CS<-TAV[J]
      TAV[J]<-TAV[J-1]
      TAV[J-1]<-CS

Ki: "A leghosszabb pont távolsága az origótól: ", TAV[1]
