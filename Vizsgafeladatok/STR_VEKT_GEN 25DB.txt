/*Az STR_VEKT_GEN szubrutin segítségével generáljon 25db 50 hosszú sztringet*/
/*1-es Seed értékkel! Határozza meg hány darab */
/*angol betu van az egyes sztringekben és a sztringeket ezen darabszámok szerint rendezze csökkeno sorrendbe! */
/*Adja meg annak a sztringnek az elso 5db karakterét amelyik ebben a sorrendben a 3. helyre került.*/
/*Mr_HumanRebel*/
N<-25
HOSSZ<-50
SEED<-1
AKTDB<-0

STR_VEKT_GEN(A,N,HOSSZ,SEED)

Ki:"Rendezetlen adatok:"

for I<-1, N
  for J<-1, HOSSZ
    if A[I][J] IN (['A'..'Z','a'..'z'])
      AKTDB<-AKTDB+1
  DB[I]<-AKTDB
  AKTDB<-0
  Ki:A[I]

Ki:"Rendezett adatok:"

for K<-1,N-1
  for X<-N,K+1,-1
    if DB[X] > DB[X-1]
      CS<-A[X]
      A[X]<-A[X-1]
      A[X-1]<-CS
  Ki:A[K]

Ki:"A harmadik helyen álló sztring elso öt karaktere:"

IDEIGLENES<- A[3]
ELSOOT<-COPY(IDEIGLENES,1,5)

Ki: ELSOOT
